```ts Example
export interface LoginOptions {
  /** The username/email to login with */
  username: string;
  /** The password for authentication */
  password: string;
  /** Optional captcha value if required */
  captcha?: string;
  /** Any additional custom options */
  [key: string]: string | number | boolean | undefined;
}
```
