---
title: "ErrorsResult"
---

An array-like object returned as part of [`UseErrorsResult`](/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/UseErrorsResult) by the [`useErrors`](/docs/libraries/acul/react-sdk/API-Reference/Hooks/useErrors) hook. Extends `ReadonlyArray<ErrorItem>`, so it can be iterated, spread, or indexed directly. Provides two methods for querying the error list by type or field.

```typescript
interface ErrorsResult extends ReadonlyArray<ErrorItem> {
  byType(type: ErrorType, opts?: { field?: string }): readonly ErrorItem[];
  byField(field: string, opts?: { type?: ErrorType }): readonly ErrorItem[];
}
```

## Methods

<ParamField body="byType" type={<span>readonly <a href="/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/ErrorItem">ErrorItem</a>[]</span>}>
  Returns all errors matching the given [`ErrorType`](/docs/libraries/acul/react-sdk/API-Reference/Types/type-aliases/ErrorType), optionally filtered to a specific field.

  <Expandable title="parameters">
    <ParamField body="type" type={<a href="/docs/libraries/acul/react-sdk/API-Reference/Types/type-aliases/ErrorType">ErrorType</a>} required>
      The error type to filter by: `"server"`, `"client"`, or `"developer"`.
    </ParamField>
    <ParamField body="opts" type="{ field?: string }">
      Optional filter options.
      <Expandable title="properties">
        <ParamField body="field" type="string">
          When provided, further filters results to errors associated with this field name.
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="byField" type={<span>readonly <a href="/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/ErrorItem">ErrorItem</a>[]</span>}>

  <Expandable title="parameters">

      The field name to filter by. Example: `"username"`, `"email"`.
    </ParamField>
    <ParamField body="opts" type={<span>&#123; type?: <a href="/docs/libraries/acul/react-sdk/API-Reference/Types/type-aliases/ErrorType">ErrorType</a> &#125;</span>}>
      Optional filter options.
      <Expandable title="properties">
        <ParamField body="type" type={<a href="/docs/libraries/acul/react-sdk/API-Reference/Types/type-aliases/ErrorType">ErrorType</a>}>

        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>
