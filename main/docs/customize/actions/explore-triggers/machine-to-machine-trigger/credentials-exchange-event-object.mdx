The `event` object for the credentials-exchange Actions trigger provides contextual information about the trigger execution.

## `event.accessToken`

<ResponseField name="event.accessToken" type="dictionary" required>
  Information about the access token to be issued.
  <Expandable title="accessToken properties" defaultOpen>
    <ResponseField name="customClaims" type="dictionary" required>
    </ResponseField>
    <ResponseField name="scope" type="array of strings" required>
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.client`

<ResponseField name="event.client" type="dictionary" required>
  Information about the Client used during this token exchange.
  <Expandable title="client properties" defaultOpen>
    <ResponseField name="client_id" type="string" required>
      The client id of the application the user is logging in to.
    </ResponseField>
    <ResponseField name="metadata" type="dictionary" required>
      An object for holding other application properties.
    </ResponseField>
    <ResponseField name="name" type="string" required>
      The name of the application (as defined in the Dashboard).
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.custom_domain`

<ResponseField name="event.custom_domain" type="dictionary">
  Details about the custom domain associated with the current transaction.
  <Expandable title="custom_domain properties" defaultOpen>
    <ResponseField name="domain" type="string" required>
      The custom domain name.
    </ResponseField>
    <ResponseField name="domain_metadata" type="dictionary" required>
      Custom domain metadata as key-value pairs.
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.organization`

<ResponseField name="event.organization" type="dictionary">
  Details about the Organization associated with the current transaction.
  <Expandable title="organization properties" defaultOpen>
    <ResponseField name="id" type="string" required>
      The Organization identifier.
    </ResponseField>
    <ResponseField name="display_name" type="string" required>
      The friendly name of the Organization.
    </ResponseField>
    <ResponseField name="metadata" type="dictionary" required>
      Metadata associated with the Organization.
    </ResponseField>
    <ResponseField name="name" type="string" required>
      The name of the Organization.
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.request`

<ResponseField name="event.request" type="dictionary" required>
  Details about the request that initiated the transaction.
  <Expandable title="request properties" defaultOpen>
    <ResponseField name="body" type="dictionary" required>
      The body of the POST request. This data will only be available during refresh token, Client Credential Exchange flows and PreUserRegistration Action.
    </ResponseField>
    <ResponseField name="geoip" type="dictionary" required>
      <Expandable title="geoip properties" defaultOpen>
        <ResponseField name="cityName" type="string">
        </ResponseField>
        <ResponseField name="continentCode" type="string">
        </ResponseField>
        <ResponseField name="countryCode" type="string">
        </ResponseField>
        <ResponseField name="countryCode3" type="string">
        </ResponseField>
        <ResponseField name="countryName" type="string">
        </ResponseField>
        <ResponseField name="latitude" type="number">
        </ResponseField>
        <ResponseField name="longitude" type="number">
        </ResponseField>
        <ResponseField name="subdivisionCode" type="string">
        </ResponseField>
        <ResponseField name="subdivisionName" type="string">
        </ResponseField>
        <ResponseField name="timeZone" type="string">
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="hostname" type="string">
      The hostname that is being used for the authentication flow.
    </ResponseField>
    <ResponseField name="ip" type="string" required>
      The originating IP address of the request.
    </ResponseField>
    <ResponseField name="language" type="string">
      The language requested by the browser.
    </ResponseField>
    <ResponseField name="method" type="string" required>
      The HTTP method used for the request
    </ResponseField>
    <ResponseField name="user_agent" type="string">
      The value of the `User-Agent` header received when initiating the transaction.
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.resource_server`

<ResponseField name="event.resource_server" type="object" required>
  Information about the Resource Server that is issuing the access token.
  <Expandable title="resource_server properties" defaultOpen>
    <ResponseField name="identifier" type="string" required>
      The identifier of the resource server. For example: `https://your-api.example.com`.
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.tenant`

<ResponseField name="event.tenant" type="object" required>
  Information about the Tenant used during this token exchange.
  <Expandable title="tenant properties" defaultOpen>
    <ResponseField name="id" type="string" required>
      The name of the tenant.
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.transaction`

<ResponseField name="event.transaction" type="object" required>
  Information about the Credentials Exchange transaction.
  <Expandable title="transaction properties" defaultOpen>
    <ResponseField name="requested_scopes" type="array of strings" required>
      The scopes specified (if any) when requesting the access token.
    </ResponseField>
  </Expandable>
</ResponseField>

## `event.secrets`

<ResponseField name="event.secrets" type="secrets" required>
  Secret values securely associated with this Action.
</ResponseField>
