The Auth0 [Management API](/docs/api/management/v2/custom-domains) provides full programmatic control over your custom domains. This guide covers common operations for managing [multiple custom domains](/docs/customize/custom-domains/multiple-custom-domains) using the Management API.

## Prerequisites

Before using the Management API with custom domains, you need:

* An Auth0 tenant on an Enterprise plan
* A Machine-to-Machine (M2M) application authorized to access the Management API
* The following permissions granted to your M2M application:
  * `read:custom_domains`
  * `create:custom_domains`
  * `update:custom_domains`
  * `delete:custom_domains`
* A valid Management API access token

To learn how to get a Management API access token, read [Get Management API Access Tokens](/docs/secure/tokens/access-tokens/management-api-access-tokens).

## Custom domain header

Many Management API operations support the `auth0-custom-domain` header, which allows you to specify which custom domain should be used for the operation:

```bash
auth0-custom-domain: your-custom-domain.com
```

### When to use the custom domain header

Use the `auth0-custom-domain` header when:

* You want to associate an operation with a specific custom domain
* You need to retrieve data in the context of a particular custom domain
* You're performing operations that affect domain-specific configurations

### Optional with default domain

If you have configured a [default custom domain](/docs/customize/custom-domains/multiple-custom-domains/default-domain), the `auth0-custom-domain` header becomes optional. When omitted, Auth0 will use your default custom domain instead of the canonical tenant domain.

## Create a custom domain

Create a new custom domain in your tenant:

<AuthCodeGroup>
```bash cURL
curl --request POST \
  --url 'https://{yourDomain}/api/v2/custom-domains' \
  --header 'authorization: Bearer {yourMgmtApiAccessToken}' \
  --header 'content-type: application/json' \
  --data '{
    "domain": "login.example.com",
    "type": "auth0_managed_certs",
    "tls_policy": "recommended",
    "custom_client_ip_header": "cf-connecting-ip",
    "domain_metadata": {
      "brand": "ExampleCorp",
      "region": "us-east",
      "environment": "production"
    }
  }'
```
```javascript Node.js
const ManagementClient = require('auth0').ManagementClient;

const management = new ManagementClient({
  domain: '{yourDomain}',
  clientId: '{yourClientId}',
  clientSecret: '{yourClientSecret}',
});

const customDomain = await management.customDomains.create({
  domain: 'login.example.com',
  type: 'auth0_managed_certs',
  tls_policy: 'recommended',
  custom_client_ip_header: 'cf-connecting-ip',
  domain_metadata: {
    brand: 'ExampleCorp',
    region: 'us-east',
    environment: 'production'
  }
});

console.log('Custom domain created:', customDomain.custom_domain_id);
```
```python Python
from auth0.management import Auth0

domain = '{yourDomain}'
mgmt_api_token = '{yourMgmtApiAccessToken}'

auth0 = Auth0(domain, mgmt_api_token)

custom_domain = auth0.custom_domains.create({
    'domain': 'login.example.com',
    'type': 'auth0_managed_certs',
    'tls_policy': 'recommended',
    'custom_client_ip_header': 'cf-connecting-ip',
    'domain_metadata': {
        'brand': 'ExampleCorp',
        'region': 'us-east',
        'environment': 'production'
    }
})

print(f"Custom domain created: {custom_domain['custom_domain_id']}")
```
```go Go
import (
    "context"
    "github.com/auth0/go-auth0/management"
)

m, err := management.New(
    domain,
    management.WithClientCredentials(clientID, clientSecret),
)
if err != nil {
    // Handle error
}

customDomain := &management.CustomDomain{
    Domain:   auth0.String("login.example.com"),
    Type:     auth0.String("auth0_managed_certs"),
    TLSPolicy: auth0.String("recommended"),
    CustomClientIPHeader: auth0.String("cf-connecting-ip"),
    DomainMetadata: map[string]interface{}{
        "brand":       "ExampleCorp",
        "region":      "us-east",
        "environment": "production",
    },
}

err = m.CustomDomain.Create(context.Background(), customDomain)
```
```java Java
import com.auth0.client.mgmt.ManagementAPI;
import com.auth0.json.mgmt.customdomains.CustomDomain;
import java.util.HashMap;
import java.util.Map;

ManagementAPI mgmt = ManagementAPI.newBuilder(
    "{yourDomain}",
    "{yourMgmtApiAccessToken}"
).build();

CustomDomain customDomain = new CustomDomain();
customDomain.setDomain("login.example.com");
customDomain.setType("auth0_managed_certs");
customDomain.setTlsPolicy("recommended");
customDomain.setCustomClientIpHeader("cf-connecting-ip");

Map<String, Object> metadata = new HashMap<>();
metadata.put("brand", "ExampleCorp");
metadata.put("region", "us-east");
metadata.put("environment", "production");
customDomain.setDomainMetadata(metadata);

CustomDomain result = mgmt.customDomains().create(customDomain).execute().getBody();
```
</AuthCodeGroup>

## List custom domains

Retrieve all custom domains configured for your tenant:

<AuthCodeGroup>
```bash cURL
curl --request GET \
  --url 'https://{yourDomain}/api/v2/custom-domains' \
  --header 'authorization: Bearer {yourMgmtApiAccessToken}'
```
```javascript Node.js
const customDomains = await management.customDomains.getAll();

customDomains.forEach(domain => {
  console.log(`${domain.domain} - Status: ${domain.status}`);
  console.log(`Metadata:`, domain.domain_metadata);
});
```
```python Python
custom_domains = auth0.custom_domains.all()

for domain in custom_domains:
    print(f"{domain['domain']} - Status: {domain['status']}")
    print(f"Metadata: {domain.get('domain_metadata', {})}")
```
```go Go
list, err := m.CustomDomain.List(context.Background())
if err != nil {
    // Handle error
}

for _, domain := range list.CustomDomains {
    fmt.Printf("%s - Status: %s\n", *domain.Domain, *domain.Status)
}
```
</AuthCodeGroup>

## Delete a custom domain

Delete a custom domain from your tenant:

<AuthCodeGroup>
```bash cURL
curl --request DELETE \
  --url 'https://{yourDomain}/api/v2/custom-domains/{customDomainId}' \
  --header 'authorization: Bearer {yourMgmtApiAccessToken}'
```
```javascript Node.js
await management.customDomains.delete({ id: '{customDomainId}' });
console.log('Custom domain deleted successfully');
```
```python Python
auth0.custom_domains.delete('{customDomainId}')
print('Custom domain deleted successfully')
```
```go Go
err = m.CustomDomain.Delete(context.Background(), customDomainID)
```
</AuthCodeGroup>

<Warning>

Deleting a custom domain is irreversible. Ensure that no applications or users are actively using the domain before deletion. Consider updating your applications to use a different custom domain before deleting.

</Warning>

## Verify a custom domain

After creating a custom domain, you must verify ownership. The verification process depends on the certificate management type:

### Auth0-managed certificates

For Auth0-managed certificates, add the provided CNAME record to your DNS:

<AuthCodeGroup>
```bash cURL
curl --request POST \
  --url 'https://{yourDomain}/api/v2/custom-domains/{customDomainId}/verify' \
  --header 'authorization: Bearer {yourMgmtApiAccessToken}'
```
```javascript Node.js
const verification = await management.customDomains.verify({
  id: '{customDomainId}'
});

if (verification.status === 'ready') {
  console.log('Domain verified successfully!');
} else {
  console.log('Verification pending. Add this CNAME record:');
  console.log(`${verification.cname_api_key}.${verification.domain} -> ${verification.origin_domain_name}`);
}
```
```python Python
verification = auth0.custom_domains.verify('{customDomainId}')

if verification['status'] == 'ready':
    print('Domain verified successfully!')
else:
    print('Verification pending. Add this CNAME record:')
    print(f"{verification['cname_api_key']}.{verification['domain']} -> {verification['origin_domain_name']}")
```
</AuthCodeGroup>

### Self-managed certificates

For self-managed certificates, provide your certificate and private key:

```bash
curl --request POST \
  --url 'https://{yourDomain}/api/v2/custom-domains/{customDomainId}/verify' \
  --header 'authorization: Bearer {yourMgmtApiAccessToken}' \
  --header 'content-type: application/json' \
  --data '{
    "type": "self_managed_certs"
  }'
```

Then add the TXT record for verification.

## Working with domain metadata

Domain metadata helps organize and filter custom domains. You can store up to 10 key-value pairs per domain.

### Use cases for metadata

* **Brand identification**: `brand: "CompanyA"`
* **Regional routing**: `region: "us-east-1"`
* **Environment tagging**: `environment: "production"`
* **Customer mapping**: `customer_id: "cust_12345"`
* **Application association**: `app_id: "app_xyz"`

### Filter domains by metadata

While the API doesn't support direct metadata filtering in list operations, you can filter client-side:

```javascript
const customDomains = await management.customDomains.getAll();

// Filter by brand
const brandDomains = customDomains.filter(
  domain => domain.domain_metadata?.brand === 'CompanyA'
);

// Filter by environment
const prodDomains = customDomains.filter(
  domain => domain.domain_metadata?.environment === 'production'
);

// Filter by region
const usEastDomains = customDomains.filter(
  domain => domain.domain_metadata?.region === 'us-east-1'
);
```

## Error handling

Handle common errors when working with the Management API:

<AuthCodeGroup>
```javascript Node.js
const ManagementClient = require('auth0').ManagementClient;

async function createCustomDomain(domainConfig) {
  try {
    const customDomain = await management.customDomains.create(domainConfig);
    return customDomain;
  } catch (error) {
    if (error.statusCode === 409) {
      console.error('Domain already exists');
    } else if (error.statusCode === 403) {
      console.error('Insufficient permissions');
    } else if (error.statusCode === 400) {
      console.error('Invalid domain configuration:', error.message);
    } else {
      console.error('Unexpected error:', error.message);
    }
    throw error;
  }
}
```
```python Python
from auth0.exceptions import Auth0Error

def create_custom_domain(auth0_client, domain_config):
    try:
        custom_domain = auth0_client.custom_domains.create(domain_config)
        return custom_domain
    except Auth0Error as e:
        if e.status_code == 409:
            print('Domain already exists')
        elif e.status_code == 403:
            print('Insufficient permissions')
        elif e.status_code == 400:
            print(f'Invalid domain configuration: {e.message}')
        else:
            print(f'Unexpected error: {e.message}')
        raise
```
</AuthCodeGroup>

## Complete example: Managing custom domains

Here's a complete example that demonstrates common operations:

```javascript
const ManagementClient = require('auth0').ManagementClient;

const management = new ManagementClient({
  domain: process.env.AUTH0_DOMAIN,
  clientId: process.env.AUTH0_CLIENT_ID,
  clientSecret: process.env.AUTH0_CLIENT_SECRET,
});

async function manageCustomDomains() {
  try {
    // 1. Create a new custom domain
    console.log('Creating custom domain...');
    const newDomain = await management.customDomains.create({
      domain: 'login.example.com',
      type: 'auth0_managed_certs',
      domain_metadata: {
        brand: 'ExampleCorp',
        region: 'us-east',
        environment: 'production'
      }
    });
    console.log('Created:', newDomain.custom_domain_id);

    // 2. Verify the domain
    console.log('Verifying domain...');
    const verification = await management.customDomains.verify({
      id: newDomain.custom_domain_id
    });

    if (verification.status !== 'ready') {
      console.log('Add this CNAME record to your DNS:');
      console.log(`${verification.cname_api_key}.${verification.domain}`);
      console.log(`  ->  ${verification.origin_domain_name}`);
      return;
    }

    // 3. Set as default domain
    console.log('Setting as default...');
    await management.customDomains.update(
      { id: newDomain.custom_domain_id },
      { is_default: true }
    );

    // 4. List all custom domains
    console.log('Listing all domains...');
    const allDomains = await management.customDomains.getAll();
    allDomains.forEach(domain => {
      const defaultBadge = domain.is_default ? '[DEFAULT]' : '';
      console.log(`${domain.domain} - ${domain.status} ${defaultBadge}`);
      console.log(`  Metadata:`, domain.domain_metadata);
    });

    // 5. Update metadata
    console.log('Updating metadata...');
    await management.customDomains.update(
      { id: newDomain.custom_domain_id },
      {
        domain_metadata: {
          ...newDomain.domain_metadata,
          last_updated: new Date().toISOString()
        }
      }
    );

    console.log('All operations completed successfully!');
  } catch (error) {
    console.error('Error:', error.message);
  }
}

manageCustomDomains();
```
