export interface ScreenMembersOnMfaSmsChallenge extends ScreenMembers {
  data: {
    /**
     * SMSが送信された電話番号
     */
    phoneNumber?: string;
    /**
     * デバイスを記憶するオプションを表示するかどうか
     */
    showRememberDevice?: boolean;
    /**
     * 音声通話認証に切り替えるリンクを表示するかどうか
     */
    showLinkVoice?: boolean;
  } | null;
}
