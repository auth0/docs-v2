````ts Example
export interface SignupMembers extends BaseMembers {
  screen: ScreenMembersOnSignup;
  transaction: TransactionMembersOnSignup;
  signup(payload: SignupOptions): Promise<void>;
  federatedSignup(payload: FederatedSignupOptions): Promise<void>;
  pickCountryCode(payload?: CustomOptions): Promise<void>;
  validatePassword(password: string): PasswordValidationResult;
  /**
 * 現在のトランザクション状態に基づき、有効な識別子（例：email、phone、username）の
 * 一覧を返します。
 *
 * 識別子が必須かオプションかは、認証またはサインアップフロー中に指定された
 * 接続戦略と設定によって異なります。
 *
 * @returns 有効な {@link Identifier} オブジェクトの配列。トランザクションが初期化されていない場合は `null`。
 *
 * @example
 * ```ts
 * const identifiers = authClient.getSignupIdentifiers();
 * if (identifiers) {
 *   identifiers.forEach(({ type, required }) => {
 *     console.log(`${type} is ${required ? 'required' : 'optional'}`);
 *   });
 * }
 * ```
 * @utilityFeature
 * @see Identifier
 */
  getSignupIdentifiers(): Identifier[] | null;
  validateUsername(username: string): UsernameValidationResult;
}
````

<ParamField body="getSignupIdentifiers" type={<span><a href="/ja/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/Identifier">Identifier</a>[]</span>}>
  **`ユーティリティ機能`**

  トランザクション状態に基づいて、有効な識別子 (例: email、phone、username) の一覧を返します。

  識別子は、接続戦略や認証またはサインアップフロー時に指定された設定に応じて、必須にも任意にもなります。

  有効な [Identifier](/ja/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/Identifier) オブジェクトの配列を返します。トランザクションが初期化されていない場合は `null` を返します。

  ```ts Example
  const identifiers = authClient.getSignupIdentifiers();
  if (identifiers) {
    identifiers.forEach(({ type, required }) => {
      console.log(`${type} is ${required ? 'required' : 'optional'}`);
    });
  }
  ```
</ParamField>
