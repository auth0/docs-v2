````ts Example
export interface MfaWebAuthnPlatformEnrollmentMembers extends BaseMembers {
  /**
   * MFA WebAuthn プラットフォーム登録固有のデータ構造を持つ画面オブジェクト。
   */
  screen: ScreenMembersOnMfaWebAuthnPlatformEnrollment;
  
  /**
   * `this.screen.publicKey` で利用可能な公開鍵オプションを使用して WebAuthn プラットフォームの認証情報作成プロセスを開始し、
   * 作成された認証情報をサーバーに送信します。
   * このメソッドは内部的に `createPasskeyCredentials`（`navigator.credentials.create()` のラッパー）を呼び出します。
   *
   * @param {SubmitPasskeyCredentialOptions} [payload] - 作成された認証情報と共にサーバーへ送信するオプションのカスタムパラメーター。
   * @returns {Promise<void>} 認証情報が正常に作成・送信されると解決される Promise。
   * @throws {Error} `this.screen.publicKey` が利用できない場合、`createPasskeyCredentials` が失敗した場合
   * （例: ユーザーによるキャンセル、ハードウェアの問題）、またはサーバーへの送信が失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnPlatformEnrollment のインスタンスと仮定
   * try {
   *   if (!sdk.screen.publicKey) { // オプションが利用可能か確認
   *     throw new Error("Public key creation options are not available on the screen context.");
   *   }
   *   await sdk.submitPasskeyCredential(); // publicKey を明示的に渡す必要はない
   *   // 成功時は Auth0 がリダイレクトを処理する。
   * } catch (error) {
   *   console.error('Passkey enrollment failed:', error);
   *   // エラーを処理する。WebAuthn API エラーの場合は reportBrowserError の呼び出しを検討する
   *   if (error.name && error.message) { // WebAuthn エラーかどうかを確認
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   * }
   * ```
   */
  submitPasskeyCredential(payload?: SubmitPasskeyCredentialOptions): Promise<void>;

  /**
   * WebAuthn の `navigator.credentials.create()` 操作中に発生したブラウザー側のエラーを報告します。
   * このメソッドはエラーの詳細をサーバーに送信します。
   *
   * @param {ReportBrowserErrorOptions} payload - ブラウザーエラーの詳細およびカスタムオプション。
   * @returns {Promise<void>} エラーレポートが正常に送信されると解決される Promise。
   * @throws {Error} 送信が失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnPlatformEnrollment のインスタンスと仮定
   * // パスキー作成の試行（例: submitPasskeyCredential から）の catch ブロック内:
   * } catch (webAuthnError) {
   *   if (webAuthnError.name) { // WebAuthn API エラーかどうかを確認
   *      await sdk.reportBrowserError({ error: { name: webAuthnError.name, message: webAuthnError.message } });
   *   }
   *   // エラーメッセージをユーザーに表示するよう UI を更新する
   * }
````

<ParamField body="submitPasskeyCredential" type="Promise<void>">
  `this.screen.publicKey` で利用可能な公開鍵オプションを使用して
  WebAuthn プラットフォーム認証情報の作成プロセスを開始し、生成された認証情報をサーバーに送信します。
  このメソッドは内部的に `createPasskeyCredentials` (`navigator.credentials.create()` をラップするメソッド) を呼び出します。

  認証情報が正常に作成および送信されると解決される Promise です。

  #### 例外

  `this.screen.publicKey` を利用できない場合、`createPasskeyCredentials` が失敗した場合
  (たとえば、ユーザーによるキャンセルやハードウェアの問題)、またはサーバーへの送信に失敗した場合はエラーをスローします。

  ```typescript Example
  // 'sdk' が MfaWebAuthnPlatformEnrollment のインスタンスであると仮定します
  try {
    if (!sdk.screen.publicKey) { // オプションが利用可能かどうかを確認します
      throw new Error("Public key creation options are not available on the screen context.");
    }
    await sdk.submitPasskeyCredential(); // publicKey を明示的に渡す必要はありません
    // 成功時は、Auth0 がリダイレクトを処理します。
  } catch (error) {
    console.error('Passkey enrollment failed:', error);
    // エラーを処理します。WebAuthn API エラーであれば、必要に応じて reportBrowserError を呼び出します
    if (error.name && error.message) { // WebAuthn エラーのように見えるかどうかを確認します
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
  }
  ```
