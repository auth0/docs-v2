---
title: "ResetPasswordMfaEmailChallengeMembers"
---

reset-password-mfa-email-challenge 画面で使用できるメソッドとプロパティを定義するインターフェース

```ts Example
export interface ResetPasswordMfaEmailChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnResetPasswordMfaEmailChallenge;

  /**
   * 指定されたcodeを使用してメールチャレンジを続行します。
   * @param payload codeとrememberDeviceフラグを含むオプション。
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * メールアドレスのコードを再送します。
   * @param payload リクエストに含める追加のカスタムオプション（省略可能）。
   */
  resendCode(payload?: ResendCodeOptions): Promise<void>;

  /**
   * ユーザーが別のMFA方式を試せるようにします。
   * @param payload リクエストに含める追加のカスタムオプション（省略可能）。
   */
  tryAnotherMethod(payload?: TryAnotherMethodOptions): Promise<void>;

  /**
   * この画面の再送機能とtimeout管理を取得します。
   * @param options 再送機能の設定オプション。
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
```
