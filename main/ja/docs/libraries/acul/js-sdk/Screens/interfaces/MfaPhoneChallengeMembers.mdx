MfaPhoneChallengeMembers

````ts Example
export interface MfaPhoneChallengeMembers extends BaseMembers {
  /**
   * @property {ScreenMembersOnMfaPhoneChallenge} screen - MFA電話チャレンジの特定の画面情報。
   */
  screen: ScreenMembersOnMfaPhoneChallenge;

  /**
   * continue
   * チャレンジを進めるためにMFA配信方法（SMSまたは音声通話）の選択を送信します。
   * これにより、ユーザーの電話に確認コードが送信されます。
   *
   * @param {ContinueOptions} payload - 配信タイプ（'sms'または'voice'）を指定するオプション。
   * @returns {Promise<void>} アクションが送信されると解決されるPromise。
   * @throws {Error} 送信に失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   *
   * // To send the code via SMS
   * await mfaPhoneChallenge.continue({ type: 'sms' });
   *
   * // To send the code via Voice Call
   * await mfaPhoneChallenge.continue({ type: 'voice' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * pickPhone
   * 別の電話番号を選択するアクションを開始します。
   * 通常、該当する場合にユーザーが複数の登録済み電話番号から選択できる画面にリダイレクトします。
   *
   * @param {PickPhoneOptions} [payload] - リクエストに含めるオプションのカスタムオプション。
   * @returns {Promise<void>} アクションが送信されると解決されるPromise。
   * @throws {Error} 送信に失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.pickPhone();
   * ```
   */
  pickPhone(payload?: PickPhoneOptions): Promise<void>;

  /**
   * tryAnotherMethod
   * 別のMFA認証要素/認証器を選択するアクションを開始します。
   * ユーザーが代替のMFA方法（OTPアプリ、リカバリーコードなど）を選択できる画面にリダイレクトします。
   *
   * @param {PickAuthenticatorOptions} [payload] - リクエストに含めるオプションのカスタムオプション。
   * @returns {Promise<void>} アクションが送信されると解決されるPromise。
   * @throws {Error} 送信に失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.tryAnotherMethod();
   * ```
   */
  tryAnotherMethod(payload?: PickAuthenticatorOptions): Promise<void>;
}

<ParamField body="continue" type="Promise<void>">
  continue
  チャレンジを続行するために、MFA の配信方法 (SMS または音声通話) を送信します。
  これにより、ユーザーの電話番号に確認コードが送信されます。

  アクションが送信されると resolve される Promise。

  #### Throws

  送信に失敗した場合は、エラーをスローします。

  ```typescript Example
  import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';

  const mfaPhoneChallenge = new MfaPhoneChallenge();

  // SMS でコードを送信する場合
  await mfaPhoneChallenge.continue({ type: 'sms' });

  // 音声通話でコードを送信する場合
  await mfaPhoneChallenge.continue({ type: 'voice' });
  ```

  <Expandable title="パラメーター">
    <ParamField body="payload" type={<span><a href="/docs/libraries/acul/js-sdk/Screens/interfaces/MfaPhoneChallengeContinueOptions">MfaPhoneChallengeContinueOptions</a></span>}>
      配信タイプ (`'sms'` または `'voice'`) を指定するオプション。

      **プロパティ**

      <ParamField body="type" type="&#x22;sms&#x22; | &#x22;voice&#x22;">
        実行するチャレンジの種類。

        * `sms`: テキストメッセージで確認コードを送信します。
        * `voice`: 音声通話で確認コードを送信します。
      </ParamField>
    </ParamField>
  </Expandable>
</ParamField>
