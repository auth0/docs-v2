---
title: "ContinuePayloadOptions"
---

メールアドレスによるチャレンジを続行するためのオプション

```ts Example
export interface ContinueOptions {
  /** ユーザーが入力したcode */
  code: string;
  /** デバイスを記憶するかどうかを示す */
  rememberDevice?: boolean;
  /** 追加のカスタムオプション */
  [key: string]: string | number | boolean | undefined;
}
```
