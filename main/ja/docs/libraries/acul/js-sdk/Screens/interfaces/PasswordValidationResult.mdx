---
title: "PasswordValidationResult"
---

```ts Example
export interface PasswordValidationResult {
  /**
   * すべてのパスワードポリシーチェックの総合結果。
   *
   * {@link PasswordComplexityRule} 内のすべての Rule の `status` が `"valid"` の場合にのみ `true`。
   */
  isValid: boolean;

  /**
   * 評価済みパスワード Rule の配列。
   *
   * 各要素は {@link PasswordComplexityRule} であり、
   * {@link PasswordComplexityRule.status | status} フィールドには
   * その Rule の検証結果が格納される。
   */
  results: PasswordComplexityRule[];
}
```

<div id="properties">
  ## プロパティ
</div>

<ParamField body="isValid" type="boolean">
  すべてのパスワードポリシーチェックの総合結果。

  [PasswordComplexityRule](/ja/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) 内のすべてのRuleの `status` が `"valid"` と等しい場合にのみ `true` になります。
</ParamField>

<ParamField body="results" type={<span><a href="/ja/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule">PasswordComplexityRule</a></span>}>
  評価されたパスワードRuleの配列。

  各項目は [PasswordComplexityRule](/ja/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) であり、その
  [status](/ja/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule#status) フィールドには、そのRuleの
  検証結果が反映されます。
</ParamField>
