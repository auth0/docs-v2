export interface ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge extends ScreenMembers {
  /**
   * Un accesseur direct pour les données `passkey.public_key` de `screen.data`.
   * Cet objet contient le challenge et les autres options nécessaires à l'appel
   * de l'API WebAuthn `navigator.credentials.get()`.
   * La valeur est `null` si `screen.data.passkey.public_key` n'est pas disponible.
   * @type {PasskeyRead['public_key'] | null}
   * @public
   */
  publicKey: PasskeyRead['public_key'] | null;

  /**
   * Un accesseur direct pour l'indicateur `show_remember_device` de `screen.data`.
   * Indique si l'interface utilisateur doit présenter à l'utilisateur une option pour mémoriser cet appareil/navigateur
   * afin de potentiellement contourner le MFA lors des connexions futures. La valeur par défaut est `false` si non défini dans `screen.data`.
   * @type {boolean}
   * @public
   */
  showRememberDevice: boolean;
}
