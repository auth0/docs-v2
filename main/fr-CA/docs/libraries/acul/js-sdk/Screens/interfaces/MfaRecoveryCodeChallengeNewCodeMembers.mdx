export interface MfaRecoveryCodeChallengeNewCodeMembers extends BaseMembers {
  /**
   * Accès aux propriétés et données spécifiques de l'écran 'mfa-recovery-code-challenge-new-code'.
   * Inclut le `textCode` qui est le code de récupération à afficher.
   * @type {ScreenMembersOnMfaRecoveryCodeChallengeNewCode}
   */
  screen: ScreenMembersOnMfaRecoveryCodeChallengeNewCode;

  /**
   * Confirme que l'utilisateur a sauvegardé le nouveau code de récupération et poursuit le flux d'authentification.
   * Cette méthode doit être appelée après que l'utilisateur indique avoir stocké de façon sécurisée le code de récupération affiché.
   * Elle envoie une confirmation au serveur Auth0 pour passer à l'étape suivante.
   *
   * @param {ContinueOptions} [payload] - Données personnalisées optionnelles à inclure dans la requête.
   * @returns {Promise<void>} Une promesse qui se résout lorsque la confirmation est soumise avec succès.
   * @throws {Error} Génère une erreur si la soumission échoue (p. ex., problème réseau, état invalide).
   *
   * @example
   * ```typescript
   * import MfaRecoveryCodeChallengeNewCode from '@auth0/auth0-acul-js/mfa-recovery-code-challenge-new-code';
   *
   * const screenManager = new MfaRecoveryCodeChallengeNewCode();
   *
   * // En supposant que l'utilisateur a coché une case confirmant qu'il a sauvegardé le code
   * try {
   *   await screenManager.continue();
   *   // Redirige vers l'écran suivant en cas de succès
   * } catch (error) {
   *   console.error("Failed to confirm recovery code saved:", error);
   *   // Gérer l'erreur, p. ex., afficher un message depuis screenManager.transaction.errors
   * }
   * ```
   */
  continue(payload?: ContinueOptions): Promise<void>;
}

<ParamField body="continue" type="Promise<void>">
  Confirme que l’utilisateur a enregistré le nouveau code de récupération et poursuit le flux d’authentification.
  Cette méthode doit être appelée après que l’utilisateur a indiqué avoir conservé de manière sécuritaire le code de récupération affiché.
  Elle envoie une confirmation au serveur Auth0 pour passer à l’étape suivante.

  Une promesse qui est résolue lorsque la confirmation est envoyée avec succès.

  #### Lève une erreur

  Lève une erreur si l’envoi échoue (p. ex., problème de réseau, `state` invalide).

  ```typescript Example
  import MfaRecoveryCodeChallengeNewCode from '@auth0/auth0-acul-js/mfa-recovery-code-challenge-new-code';

  const screenManager = new MfaRecoveryCodeChallengeNewCode();

  // Supposons que l’utilisateur a coché une case pour confirmer qu’il a enregistré le code
  try {
    await screenManager.continue();
    // Redirige vers l’écran suivant en cas de succès
  } catch (error) {
    console.error("Failed to confirm recovery code saved:", error);
    // Traitez l’erreur, p. ex., en affichant un message provenant de screenManager.transaction.errors
  }
  ```
