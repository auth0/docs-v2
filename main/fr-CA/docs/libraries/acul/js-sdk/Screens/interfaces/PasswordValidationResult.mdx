title: "PasswordValidationResult"

```ts Example
export interface PasswordValidationResult {
  /**
   * Résultat global de toutes les vérifications de la stratégie de mot de passe.
   *
   * `true` uniquement si chaque Rule dans {@link PasswordComplexityRule} a un `status` égal à `"valid"`.
   */
  isValid: boolean;

  /**
   * Tableau des Rules de mot de passe évaluées.
   *
   * Chaque élément est un {@link PasswordComplexityRule} dont
   * le champ {@link PasswordComplexityRule.status | status} reflète le
   * résultat de validation calculé pour cette Rule.
   */
  results: PasswordComplexityRule[];
}
```
