title: Plantillas de scripts de eliminación
---

El script Delete implementa la función definida para eliminar un usuario especificado de una base de datos externa. Recomendamos llamar a esta función `deleteUser`.

El script solo se usa en un [escenario de autenticación heredado](/es/docs/authenticate/database-connections/custom-db/overview-custom-db-connections) y es necesario si quiere eliminar un usuario de Auth0 y de su base de datos externa en la misma operación.

<div id="deleteuser-function">
  ## Función DeleteUser
</div>

La función `deleteUser` debe:

* Eliminar el usuario especificado de la base de datos externa.
* Devolver un valor `null` si la eliminación se realiza correctamente.
* Devolver un error si la eliminación falla.

<div id="definition">
  ### Definición
</div>

La función `deleteUser` recibe dos parámetros y devuelve una función `callback`:

```js lines
deleteUser(id, callback): function
```

| Parámetro  | Tipo     | Descripción                                                      |
| ---------- | -------- | ---------------------------------------------------------------- |
| `id`       | String   | `user_id` de identidad del usuario. No comienza con `auth0`.     |
| `callback` | Function | Se usa para pasar errores o datos del perfil a través del flujo. |

### Ejemplo

Este es un ejemplo en seudo-JavaScript de cómo podría implementar la función `deleteUser`. Para ver ejemplos específicos de cada lenguaje, consulte [Ejemplos de scripts específicos de cada lenguaje](/es/docs/authenticate/database-connections/custom-db/templates/delete#language-specific-script-examples).

```javascript lines
function deleteUser(id, callback) {
  // Enviar el identificador de usuario a la API de base de datos externa
  let options = {
    url: "https://example.com/api/deleteUser",
    body: {
      id: id
    }
  };

  send(options, (err, profileData) => {
    // Devolver error en el callback si la eliminación falló
    if (err) {
      return callback(new Error("My custom error message."));
    }

    // Devolver valor nulo en el callback si la eliminación fue exitosa
    return callback(null);
  });
}
```

<div id="callback-function">
  ## Función de devolución de llamada
</div>

La función `callback` acepta un parámetro y devuelve una función.

<div id="definition">
  ### Definición
</div>

```js lines
callback(error): function
```

| Parámetro | Tipo   | Obligatorio | Descripción                   |
| --------- | ------ | ----------- | ----------------------------- |
| `error`   | Objeto | Obligatorio | Contiene los datos del error. |

<div id="return-a-success">
  ### Devolver un resultado exitoso
</div>

Si el usuario se eliminó correctamente de su base de datos externa, devuelva la función `callback` y pase un valor `null` en el parámetro `error`.

<div id="example">
  #### Ejemplo
</div>

```js lines
return callback(null);
```

<div id="return-an-error">
  ### Devolver un error
</div>

Si se produce un error, llama a la función `callback` y pasa la información relevante del error al parámetro `error`.

<div id="example">
  #### Ejemplo
</div>

```js lines
return callback(new Error("My custom error message."));
```

<div id="javascript">
  ### JavaScript
</div>

```javascript lines
function remove (id, callback) {
  // Este script elimina un usuario de su base de datos existente.
  // Se ejecuta cada vez que se elimina un usuario desde la Management API o el Auth0 Dashboard.
  //
  // Hay dos formas en que este script puede finalizar:
  // 1. El usuario fue eliminado correctamente:
  //     callback(null);
  // 2. Algo salió mal al intentar conectarse a su base de datos:
  //     callback(new Error("my error message"));
  var msg = "Please implement the Delete script for this database " +
       "connection at https://manage.auth0.com/#/connections/database";
  return callback(new Error(msg));
}
```

<div id="aspnet-membership-provider-mvc3-universal-providers">
  ### Proveedor de membresía de ASP.NET (MVC3 - Proveedores universales)
</div>

```javascript lines expandable
function remove(id, callback) {
  const sqlserver = require('tedious@1.11.0');
  const Connection = sqlserver.Connection;
  const Request = sqlserver.Request;
  const TYPES = sqlserver.TYPES;
  const connection = new Connection({
    userName: 'the username',
    password: 'the password',
    server: 'the server',
    options: {
      database: 'the db name',
      encrypt: true,
      // Necesario para recuperar el userId requerido para la creación de la entidad Membership
      rowCollectionOnRequestCompletion: true
    }
  });
  connection.on('debug', function(text) {
    // si tienes problemas de conexión, descomenta esto para obtener información más detallada
    // console.log(text);
  }).on('errorMessage', function(text) {
    // esto mostrará cualquier error al conectarse a la base de datos SQL o con las sentencias SQL
    console.log(JSON.stringify(text));
  });
  connection.on('connect', function(err) {
    if (err) return callback(err);
    executeDelete(['Memberships', 'Users'], function(err) {
      if (err) return callback(err);
      callback(null);
    });
  });
  function executeDelete(tables, callback) {
    const query = tables.map(function(table) {
      return 'DELETE FROM ' + table + ' WHERE UserId = @UserId';
    }).join(';');
    const request = new Request(query, function(err) {
      if (err) return callback(err);
      callback(null);
    });
    request.addParameter('UserId', TYPES.VarChar, id);
    connection.execSql(request);
  }
}
```

<div id="mongodb">
  ### MongoDB
</div>

```javascript lines
function remove(id, callback) {
  const MongoClient = require('mongodb@3.1.4').MongoClient;
  const client = new MongoClient('mongodb://user:pass@mymongoserver.com');
  client.connect(function (err) {
    if (err) return callback(err);
    const db = client.db('db-name');
    const users = db.collection('users');
    users.remove({ _id: id }, function (err) {
      client.close();
      if (err) return callback(err);
      callback(null);
    });
  });
}
```

<div id="postgresql">
  ### PostgreSQL
</div>

```javascript lines
function remove(id, callback) {
  //este ejemplo usa la librería "pg"
  //más información aquí: https://github.com/brianc/node-postgres
  const postgres = require('pg');
  const conString = 'postgres://user:pass@localhost/mydb';
  postgres.connect(conString, function (err, client, done) {
    if (err) return callback(err);
    const query = 'DELETE FROM users WHERE id = $1';
    client.query(query, [id], function (err) {
      // NOTA: siempre llama a `done()` aquí para cerrar
      // la conexión a la base de datos
      done();
      return callback(err);
    });
  });
}
```

  // este ejemplo usa la biblioteca "tedious"
  // más información aquí: http://pekim.github.io/tedious/index.html
