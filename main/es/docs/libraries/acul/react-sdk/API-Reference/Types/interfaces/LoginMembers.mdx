---
title: "LoginMembers"
---

Interfaz de miembros de la pantalla Login que extiende la interfaz base de miembros

```ts Example
export interface LoginMembers extends BaseMembers {
  screen: ScreenMembersOnLogin;
  transaction: TransactionMembersOnLogin;
  /**
   * Realiza el login con username/password
   * @param payload Las opciones de login
   */
  login(payload: LoginOptions): Promise<void>;
  /**
   * Realiza el login con proveedor social
   * @param payload Las opciones de login social
   */
  federatedLogin(payload: FederatedLoginOptions): Promise<void>;
  /**
   * Selecciona el código de país para la entrada del número de teléfono
   * @param payload Opciones personalizadas adicionales opcionales
   */
  pickCountryCode(payload?: CustomOptions): Promise<void>;
  /**
   * Obtiene los tipos de identificador activos para la pantalla de login
   * @returns Un array de tipos de identificador activos o null si no hay ninguno activo
   * @utilityFeature
   */
  getLoginIdentifiers(): IdentifierType[] | null;
}
```
