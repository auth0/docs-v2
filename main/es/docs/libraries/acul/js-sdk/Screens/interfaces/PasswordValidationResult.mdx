title: "PasswordValidationResult"
---

```ts Example
export interface PasswordValidationResult {
  /**
   * Resultado general de todas las verificaciones de política de contraseña.
   *
   * `true` solo si cada Rule en {@link PasswordComplexityRule} tiene `status` igual a `"valid"`.
   */
  isValid: boolean;

  /**
   * Array de Rules de contraseña evaluadas.
   *
   * Cada elemento es un {@link PasswordComplexityRule} cuyo
   * campo {@link PasswordComplexityRule.status | status} refleja el
   * resultado de validación calculado para esa Rule.
   */
  results: PasswordComplexityRule[];
}
```

<div id="properties">
  ## Propiedades
</div>

<ParamField body="isValid" type="boolean">
  Resultado global de todas las comprobaciones de la política de contraseñas.

  `true` solo si cada Rule de [PasswordComplexityRule](/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) tiene `status` igual a `"valid"`.
</ParamField>

<ParamField body="results" type={<span><a href="/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule">PasswordComplexityRule</a></span>}>
  Array de Rules de contraseña evaluadas.

  Cada elemento es una [PasswordComplexityRule](/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) cuyo
  campo [status](/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule#status) refleja el
  resultado calculado de la validación de esa Rule.
</ParamField>
